## 1. Docker Compose 编排

```yaml
# docker-compose.yml

services:

      - "8080:8080"
    environment:

        condition: service_healthy

        condition: service_healthy

        condition: service_healthy

      - milvus-data:/var/lib/milvus

      ETCD_ENDPOINTS: etcd:2379

      test: ["CMD", "curl", "-f", "http://localhost:9091/healthz"]

      - etcd
      - minio

    image: minio/minio:RELEASE.2024-05-10T01-15-34Z

      MINIO_ACCESS_KEY: minioadmin
      MINIO_SECRET_KEY: minioadmin

      - minio-data:/minio_data
    command: minio server /minio_data --console-address ":9001"
    ports:
      - "9001:9001"
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:9000/minio/health/live"]
      interval: 10s
      timeout: 5s
      retries: 3
    restart: unless-stopped

  # MySQL
  mysql:
    image: mysql:8.0
    ports:
      - "3306:3306"
    environment:
      MYSQL_ROOT_PASSWORD: root123
      MYSQL_DATABASE: offerpilot
      MYSQL_CHARACTER_SET_SERVER: utf8mb4
      MYSQL_COLLATION_SERVER: utf8mb4_unicode_ci
    volumes:
      - mysql-data:/var/lib/mysql
    healthcheck:
      test: ["CMD", "mysqladmin", "ping", "-h", "localhost"]
      interval: 10s
      timeout: 5s
      retries: 5
    restart: unless-stopped

  # Redis 缓存（会话缓存 + 热点数据 + 限流）

    image: redis:7-alpine
    ports:
      - "6379:6379"
    volumes:
      - redis-data:/data
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 10s
      timeout: 3s
      retries: 3
    restart: unless-stopped

volumes:
  milvus-data:
  etcd-data:
```

### 部署步骤

```bash
# 1. 创建 .env 文件
echo "DASHSCOPE_API_KEY=sk-xxx" > .env
echo "JWT_SECRET=$(openssl rand -base64 32)" >> .env

# 2. 启动所有服务
docker-compose up -d

# 3. 检查服务状态
docker-compose ps

# 4. 查看应用日志
docker-compose logs -f offerpilot-app

# 5. 停止服务
docker-compose down
```

## 2. 错误处理策略

### 2.1 异常体系

```java
/** 业务异常基类 — 所有自定义异常的父类 */
@Getter
public abstract class BusinessException extends RuntimeException {
    private final String errorCode;
    private final int httpStatus;

    protected BusinessException(String errorCode, int httpStatus, String message) {
        super(message);
        this.errorCode = errorCode;
        this.httpStatus = httpStatus;
    }
}

/** 409 — 用户名已存在 */
public class DuplicateUserException extends BusinessException {
    public DuplicateUserException(String username) {
        super("DUPLICATE_USER", 409, "用户名 '" + username + "' 已存在");
    }
}

/** 401 — 密码错误 */
public class InvalidCredentialsException extends BusinessException {
    public InvalidCredentialsException() {
        super("INVALID_CREDENTIALS", 401, "用户名或密码错误");
    }
}

/** 403 — 账号已禁用 */
public class AccountDisabledException extends BusinessException {
    public AccountDisabledException() {
        super("ACCOUNT_DISABLED", 403, "账号已被禁用，请联系管理员");
    }
}

/** 429 — 请求过于频繁 */
public class RateLimitException extends BusinessException {
    public RateLimitException(String message) {
        super("RATE_LIMITED", 429, message);
    }
}

/** 504 — Agent 超时 */
public class AgentTimeoutException extends BusinessException {
    public AgentTimeoutException(String message) {
        super("AGENT_TIMEOUT", 504, message);
    }
}
```

### 2.2 统一响应包装

```java
/** 统一响应体 — 所有接口返回此格式 */
@Data
@AllArgsConstructor
public class ApiResponse<T> {
    private int code;
    private String message;
    private T data;

    public static <T> ApiResponse<T> success(T data) {
        return new ApiResponse<>(200, "success", data);
    }

    public static <T> ApiResponse<T> error(int code, String message) {
        return new ApiResponse<>(code, message, null);
    }
}
```

### 2.3 全局异常处理器

```java
@RestControllerAdvice
public class GlobalExceptionHandler {

    /** 业务异常 — 根据 httpStatus 动态返回状态码 */
    @ExceptionHandler(BusinessException.class)
    public ResponseEntity<ApiResponse<Void>> handleBusiness(BusinessException e) {
        return ResponseEntity.status(e.getHttpStatus())
            .body(ApiResponse.error(e.getHttpStatus(), e.getMessage()));
    }

    /** 参数校验失败（@Valid 触发） */
    @ExceptionHandler(MethodArgumentNotValidException.class)
    public ResponseEntity<ApiResponse<Map<String, String>>> handleValidation(
            MethodArgumentNotValidException e) {
        Map<String, String> fieldErrors = new HashMap<>();
        e.getBindingResult().getFieldErrors()
            .forEach(err -> fieldErrors.put(err.getField(), err.getDefaultMessage()));
        return ResponseEntity.badRequest()
            .body(new ApiResponse<>(400, "参数校验失败", fieldErrors));
    }

    /** Spring Security 认证异常 */
    @ExceptionHandler(AuthenticationException.class)
    public ResponseEntity<ApiResponse<Void>> handleAuth(AuthenticationException e) {
        return ResponseEntity.status(401)
            .body(ApiResponse.error(401, "认证失败：" + e.getMessage()));
    }

    /** Spring Security 授权异常 */
    @ExceptionHandler(AccessDeniedException.class)
    public ResponseEntity<ApiResponse<Void>> handleAccessDenied(AccessDeniedException e) {
        return ResponseEntity.status(403)
            .body(ApiResponse.error(403, "权限不足"));
    }

    /** Milvus 向量库异常 */
    @ExceptionHandler(MilvusException.class)
    public ResponseEntity<ApiResponse<Void>> handleMilvus(MilvusException e) {
        log.error("Milvus error: {}", e.getMessage(), e);
        return ResponseEntity.status(503)
            .body(ApiResponse.error(503, "向量数据库服务不可用"));
    }

    /** 兜底 — 未预期的异常 */
    @ExceptionHandler(Exception.class)
    public ResponseEntity<ApiResponse<Void>> handleGeneral(Exception e) {
        log.error("Unexpected error", e);
        return ResponseEntity.status(500)
            .body(ApiResponse.error(500, "服务器内部错误，请联系管理员"));
    }
}
```

**设计要点：**
- 所有接口统一返回 `ApiResponse<T>` 格式 `{code, message, data}`
- `BusinessException` 基类携带 `httpStatus`，`GlobalExceptionHandler` 按状态码动态响应
- `AuthService` 中不再用 `IllegalArgumentException`，改为 `DuplicateUserException` / `InvalidCredentialsException`
- `@Valid` 校验失败自动返回字段级错误明细

### 2.4 审计基类 BaseEntity

```java
/** 所有 JPA 实体继承此类，自动填充创建/更新时间 */
@MappedSuperclass
@EntityListeners(AuditingEntityListener.class)
@Data
public abstract class BaseEntity {

    @CreatedDate
    @Column(updatable = false)
    private LocalDateTime createdAt;

    @LastModifiedDate
    private LocalDateTime updatedAt;
}
```

启动类需加 `@EnableJpaAuditing` 注解，实体继承示例：

```java
@Entity
@Table(name = "op_app_user")
public class AppUser extends BaseEntity {
    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    @Column(unique = true, nullable = false)
    private String userId;
    // ... 其他字段
}
```

### 2.5 枚举类（消除硬编码字符串）

```java
public enum UserRole {

public enum InterviewMode {
    TECHNICAL, BEHAVIORAL, SYSTEM_DESIGN, PRESSURE

public enum MemoryCategory {
    PROFILE, WEAK_POINT, PREFERENCE, PLAN, GENERAL

/** 会话状态 */
public enum SessionStatus {
    ACTIVE, COMPLETED, ABANDONED
}
```

实体中使用枚举字段：

```java
@Entity
@Table(name = "op_kb_knowledge_base")
public class KbKnowledgeBase extends BaseEntity {

    @Enumerated(EnumType.STRING)       // 存数据库为字符串，避免 ordinal 漂移
    @Column(nullable = false)
    private Visibility visibility;

    @Enumerated(EnumType.STRING)
    @Column(nullable = false)
    private DocumentStatus status;
}
```

### 2.6 DTO 校验注解

```java
/** 注册请求 */
@Data
public class RegisterRequest {
    @NotBlank(message = "用户名不能为空")
    @Size(min = 3, max = 32, message = "用户名长度 3-32 位")
    private String username;

    @NotBlank(message = "密码不能为空")
    @Size(min = 6, max = 64, message = "密码长度 6-64 位")
    private String password;

    @Email(message = "邮箱格式不正确")
    private String email;
}

/** 登录请求 */
@Data
public class LoginRequest {
    @NotBlank(message = "用户名不能为空")
    private String username;

    @NotBlank(message = "密码不能为空")
    private String password;
}

/** 对话请求 */
@Data
public class ChatRequest {
    @NotBlank(message = "消息内容不能为空")
    private String message;

    private String sessionId;           // 可选，为空时自动创建
}

/** 创建知识库请求 */
@Data
public class CreateKbRequest {
    @NotBlank(message = "知识库名称不能为空")
    @Size(max = 64)
    private String name;

    @Size(max = 256)
    private String description;

    private String category;
}
```

Controller 中配合 `@Valid` 使用，校验失败由 `GlobalExceptionHandler.handleValidation()` 统一处理，返回字段级错误信息。
